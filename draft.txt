// Async validation example
const schema3 = createForme({
  username: {
    // ...
    validations: {
      // ...
      isUnique: {
        message: "This username already exists",
        handler: async (value) => {
          let request = await axios.post('@api/username-is-unique', { username: value });
          return request.json();
        }
      }
    }
  }
  // ...
});


// Input event handling handling

      console.log(schema) // reference to the reactive schema
    }  
  }
  // ...
});
